 43        function setupImpl(obj,u)
 44            % Implement tasks that need to be performed only once, 
 45            % such as pre-computed constants.
 46            obj.Delta1 = 2*rand(1)-1;
 47            obj.Delta2 = 2*rand(1)-1;
 48            obj.R       = 6.2;                  % Resistance induit
 49            obj.L       = 0.8e-3;               % Inductance induit
 50            obj.J1      = 0.039e-4;             % Inertie rotor
 51            obj.TauM1   = 19.6e-3;              % Constante de temps mecanique rotor seul
 52            obj.Ke      = 3.6/1000*60/(2*pi);   % Constante de fem
 53            obj.Kc      = 3.5/100;              % Constante de couple
 54            obj.Ks      = 10;                   % Constante du potentiometre
 55            obj.Kr      = 1/9;                  % Facteur du reducteur
 56            obj.Cn      = 3e-4;                 % Couple nominal de frottement sec si rotation
 57            obj.rCn     = obj.Cn/2;                 % Rayon de l'incertitude du frottement sec
 58            obj.Kg      = 0.105;                % Constante de la generatrice tachymetrique
 59            obj.Rchn    = 100;                  % Résistance nominale de la charge de la génératrice tachymétrique
 60            obj.rRch    = 0.5;                  % Rayon de l'incertitude de la résistance de charge
 61            obj.Znmrt   = 0.01;
 62            obj.TauM2   = 500e-3;               % Constante de temps mecanique rotor+reducteur+charge
 63            obj.mu      = obj.J1/obj.TauM1;             % Coefficient de frottement visqueux
 64            obj.J2      = obj.TauM2*obj.mu;             % Intertie rotor+reducteur+charge
 65

 67            obj.xC = zeros(4,1);

 70        function y = stepImpl(obj,u)%u=Vm
 71            %f
 72            % Implement algorithm. Calculate y as a function of
 73            % input u and discrete states.
 74            obj.dx(1) = -obj.R/obj.L*obj.xC(1) - obj.Ke/obj.L*obj.xC(3) + 1/obj.L*u;
 75            obj.dx(2) = -(obj.R+(obj.Rchn+obj.rRch*obj.Delta2))/obj.L*obj.xC(2) - obj.Ke/obj.L*obj.xC(3);
 76
 77            if(abs(obj.xC(3))>obj.Znmrt)
 78                obj.dx(3) = obj.Kc/obj.J2*obj.xC(1) + obj.Kc/obj.J2*obj.xC(2) - obj.mu/obj.J2*obj.xC(3) - sign(obj.xC(3))/obj.J2*obj.Cn*(1+obj.rCn*obj.Delta1);
 79            else
 80                if(abs(obj.Kc*obj.xC(1) + obj.Kc*obj.xC(2) - obj.mu*obj.xC(3)) >= obj.Cn*(1+obj.rCn*obj.Delta1))
 81                    obj.dx(3) = obj.Kc/obj.J2*obj.xC(1) + obj.Kc/obj.J2*obj.xC(2) - obj.mu/obj.J2*obj.xC(3) - sign(obj.Kc*obj.xC(1) + obj.Kc*obj.xC(2) - obj.mu*obj.xC(3))/obj.J2*obj.Cn*(1+obj.rCn*obj.Delta1);
 82                else
 83                    obj.xC(3)=0;
 84                    obj.dx(3)=0;
 85                end
 86            end
 87            obj.dx(4) = obj.xC(3);
 88
 89            %m
 90            for i = 1:3
 91                obj.xd(i)=obj.xd(i+1);
 92                obj.xC(i)=obj.xC(i+1);
 93            end
 94            % Outputs
 95            y = zeros(2,1);
 96            y(1)= obj.Kr*obj.Ks*obj.xC(4)-10*floor((obj.Kr*obj.Ks*obj.xC(4)+5)/10);%Vs
 97            y(2)= obj.Kg*obj.xC(3);%Vg
 98            
 99        end
